names_of_company = ['HBL']
HBL = pd.DataFrame()
for names in names_of_company:
    x = quandl.get('PSX/%s' %names, start_date='2011-1-1', end_date='2021-7-30')
    x['company'] = names
    HBL = HBL.append(x)
    
HBL = HBL.reset_index
HBL = HBL('Date')
HBL_Turnover = HBL['Turnover']
HBL_date = HBL['Date']
HBL_daily = HBL['Open'] - HBL['Close'].shift(1)
HBL_monthMean = HBL.resample('M', on='Date').mean()
HBL_yearMean = HBL.resample('Y', on='Date').mean()
HBL_dailyReturn = (HBL['Close'].shift(1) / HBL['Close']) - 1
